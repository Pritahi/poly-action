   326→// Login
   327→async function doLogin() {
   328→  const u = document.getElementById('login-user').value;
   329→  const p = document.getElementById('login-pass').value;
   330→  try {
   331→    const r = await fetch(API + '/api/admin/login', {
   332→      method: 'POST', headers: {'Content-Type':'application/json'},
   333→      body: JSON.stringify({username: u, password: p})
   334→    });
   335→    if (!r.ok) { document.getElementById('login-error').style.display='block'; return; }
   336→    document.getElementById('login-screen').style.display='none';
   337→    document.getElementById('app-content').style.display='block';
   338→    loadAll();
   339→  } catch(e) { document.getElementById('login-error').style.display='block'; }
   340→}
   341→
   342→// Check session on load
   343→(async () => {
   344→  try {
   345→    const r = await fetch(API + '/api/admin/me');
   346→    if (r.ok) {
   347→      document.getElementById('login-screen').style.display='none';
   348→      document.getElementById('app-content').style.display='block';
   349→      loadAll();
   350→    }
   351→  } catch(e) {}
   352→})();

   368→// Stats
   369→async function loadStats() {
   370→  const [statsRes, usersRes] = await Promise.all([
   371→    fetch(API + '/api/admin/stats'),
   372→    fetch(API + '/api/admin/users?per_page=100')
   373→  ]);
   374→  const stats = await statsRes.json();
   375→  const usersData = await usersRes.json();
   376→
   377→  // Stats cards
   378→  const cards = [
   379→    {label:'Total Users', value: stats.total_users, color:'var(--accent)'},
   380→    {label:'Active Users', value: stats.active_users, color:'var(--green)'},
   381→    {label:'New This Week', value: stats.new_this_week, color:'var(--blue)'},
   382→    {label:'New Today', value: stats.new_today, color:'var(--yellow)'},
   383→    {label:'Pro Users', value: stats.pro_users, color:'var(--accent2)'},
   384→    {label:'Enterprise', value: stats.enterprise_users, color:'var(--orange)'},
   385→    {label:'Total Repos', value: stats.total_repos, color:'var(--green)'},
   386→    {label:'CI Runs Today', value: stats.runs_today, color:'var(--blue)'},
   387→  ];
   388→  document.getElementById('stats-grid').innerHTML = cards.map(c =>
   389→    `<div class="stat-card"><div class="label">${c.label}</div><div class="value" style="color:${c.color}">${c.value}</div></div>`
   390→  ).join('');
   391→
   392→  // Signup chart

   440→async function loadUsers() {
   441→  const search = document.getElementById('search-input').value;
   442→  const plan = document.getElementById('plan-filter').value;
   443→  const sort = document.getElementById('sort-select').value;
   444→  const res = await fetch(`${API}/api/admin/users?search=${encodeURIComponent(search)}&plan=${plan}&sort=${sort}&page=${currentPage}&per_page=15`);

   452→      </td>
   453→      <td><span class="tag tag-${u.plan}">${u.plan}</span></td>
   454→      <td>${u.signup_source ? '<span class="tag tag-source">'+esc(u.signup_source)+'</span>' : '<span style="color:var(--muted)">-</span>'}</td>
   455→      <td style="color:var(--muted);font-size:12px">${esc(u.referrer || '-')}</td>

   460→      <td><span class="tag tag-${u.is_active ? 'active' : 'inactive'}">${u.is_active ? 'Active' : 'Inactive'}</span></td>
   461→      <td style="color:var(--muted);font-size:12px;white-space:nowrap">${formatDate(u.created_at)}</td>
   462→      <td>
   463→        <button class="btn btn-sm" onclick="editUser(${u.id},'${esc(u.name)}','${esc(u.email||'')}','${esc(u.github_username||'')}','${u.plan}','${esc(u.signup_source||'')}','${esc(u.referrer||'')}','${esc(u.notes||'')}')">Edit</button>
   464→        <button class="btn btn-sm btn-danger" onclick="deleteUser(${u.id},'${esc(u.name)}')">Del</button>
   465→      </td>
   466→    </tr>
   467→  `).join('') : '<tr><td colspan="8" style="text-align:center;color:var(--muted);padding:32px">No users found</td></tr>';
   468→
   469→  // Pagination
   470→  const pages = data.total_pages || 1;
   471→  let pagHtml = `<button ${currentPage<=1?'disabled':''} onclick="goPage(${currentPage-1})">Prev</button>`;
   472→  for (let i = 1; i <= pages; i++) {
   473→    if (pages > 7 && Math.abs(i - currentPage) > 2 && i !== 1 && i !== pages) {
   474→      if (i === currentPage - 3 || i === currentPage + 3) pagHtml += '<button disabled>...</button>';
   475→      continue;
   476→    }
   477→    pagHtml += `<button class="${i===currentPage?'active':''}" onclick="goPage(${i})">${i}</button>`;
   478→  }
   479→  pagHtml += `<button ${currentPage>=pages?'disabled':''} onclick="goPage(${currentPage+1})">Next</button>`;
   480→  document.getElementById('pagination').innerHTML = pagHtml;
   481→}

   515→async function saveUser() {
   516→  const id = document.getElementById('edit-user-id').value;
   517→  const body = {
   518→    name: document.getElementById('form-name').value,
   519→    email: document.getElementById('form-email').value,
   520→    github_username: document.getElementById('form-github').value || null,
   521→    plan: document.getElementById('form-plan').value,
   522→    signup_source: document.getElementById('form-source').value || null,
   523→    referrer: document.getElementById('form-referrer').value || null,
   524→    notes: document.getElementById('form-notes').value || null,
   525→  };
   526→  if (!body.name || !body.email) { toast('Name and email required', 'error'); return; }
   527→
   528→  let res;
   529→  if (id) {
   530→    res = await fetch(`${API}/api/admin/users/${id}`, {method:'PUT', headers:{'Content-Type':'application/json'}, body: JSON.stringify(body)});
   531→  } else {
   532→    res = await fetch(`${API}/api/admin/users`, {method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify(body)});
   533→  }
   534→  if (!res.ok) { const e = await res.json(); toast(e.detail || 'Error', 'error'); return; }
   535→  closeModal();
   536→  toast(id ? 'User updated' : 'User created', 'success');
   537→  loadAll();
   538→}

   560→// Helpers
   561→function esc(s) { return s ? s.replace(/'/g, "\\'").replace(/"/g, '&quot;') : ''; }
   562→function formatDate(d) { if (!d) return '-'; return new Date(d + 'Z').toLocaleDateString('en-US', {month:'short',day:'numeric',year:'numeric'}); }
   563→function timeAgo(d) {
   564→  if (!d) return '';
   565→  const s = Math.floor((Date.now() - new Date(d + 'Z').getTime()) / 1000);
   566→  if (s < 60) return 'just now';
   567→  if (s < 3600) return Math.floor(s/60) + 'm ago';
   568→  if (s < 86400) return Math.floor(s/3600) + 'h ago';
   569→  return Math.floor(s/86400) + 'd ago';
   570→}
