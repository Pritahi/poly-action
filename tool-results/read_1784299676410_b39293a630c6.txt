   209тЖТ   209тЖТdef _beta_pdf(x, alpha, beta_param):
   210тЖТ   210тЖТ    """Approximate Beta PDF value (log-space for numerical stability)."""
   211тЖТ   211тЖТ    if x <= 0 or x >= 1:
   212тЖТ   212тЖТ        return 0.0
   213тЖТ   213тЖТ    # log(B(alpha, beta)) = log(Gamma(alpha)) + log(Gamma(beta)) - log(Gamma(alpha+beta))
   214тЖТ   214тЖТ    # Using Stirling's approximation for ln(Gamma)
   215тЖТ   215тЖТ    def ln_gamma(z):
   216тЖТ   216тЖТ        if z < 0.5:
   217тЖТ   217тЖТ            return math.log(math.pi / math.sin(math.pi * z)) - ln_gamma(1 - z)
   218тЖТ   218тЖТ        z -= 1
   219тЖТ   219тЖТ        g = 0.99999999999980993 + (
   220тЖТ   220тЖТ            676.5203681218851 / (z + 1) - 1259.1392167224028 / (z + 2) +
   221тЖТ   221тЖТ            771.32342877765313 / (z + 3) - 176.61502916214059 / (z + 4) +
   222тЖТ   222тЖТ            12.507343278686905 / (z + 5) - 0.13857109526572012 / (z + 6) +
   223тЖТ   223тЖТ            9.9843695780195716e-6 / (z + 7) + 1.5056327351493116e-7 / (z + 8)
   224тЖТ   224тЖТ        )
   225тЖТ   225тЖТ        return 0.5 * math.log(2 * math.pi) + (z + 0.5) * math.log(z + 7.5) - (z + 7.5) + math.log(g)
   226тЖТ   226тЖТ
   227тЖТ   227тЖТ    log_beta = ln_gamma(alpha) + ln_gamma(beta_param) - ln_gamma(alpha + beta_param)
   228тЖТ   228тЖТ    log_pdf = (alpha - 1) * math.log(x) + (beta_param - 1) * math.log(1 - x) - log_beta
   229тЖТ   229тЖТ    return math.exp(log_pdf)

   232тЖТ   232тЖТdef calculate_bayesian_pass_rate(history, prior_alpha=2.0, prior_beta=2.0):
   233тЖТ   233тЖТ    """
   234тЖТ   234тЖТ    Bayesian pass rate using Beta-Binomial conjugate prior.
   235тЖТ   235тЖТ    
   236тЖТ   236тЖТ    Instead of naive pass/total, we use:
   237тЖТ   237тЖТ    - Prior: Beta(2, 2) тАФ slight bias toward stability (most tests should pass)
   238тЖТ   238тЖТ    - Posterior: Beta(2 + passes, 2 + failures)
   239тЖТ   239тЖТ    - MAP estimate: (alpha - 1) / (alpha + beta - 2)
   240тЖТ   240тЖТ    
   241тЖТ   241тЖТ    This means:
   242тЖТ   242тЖТ    - 0/0 runs тЖТ 50% (prior mean) instead of undefined
   243тЖТ   243тЖТ    - 1/1 run тЖТ 60% (pulled toward prior) instead of 100%
   244тЖТ   244тЖТ    - 10/10 runs тЖТ 92% (prior has less influence with more data)
   245тЖТ   245тЖТ    - 5/10 runs тЖТ 46% (properly uncertain)
   246тЖТ   246тЖТ    """
   247тЖТ   247тЖТ    if not history:
   248тЖТ   248тЖТ        return 0.5  # Prior mean
   249тЖТ   249тЖТ
   250тЖТ   250тЖТ    passed = sum(1 for h in history if h["status"] == "passed")
   251тЖТ   251тЖТ    failed = len(history) - passed
   252тЖТ   252тЖТ
   253тЖТ   253тЖТ    alpha = prior_alpha + passed
   254тЖТ   254тЖТ    beta_p = prior_beta + failed
   255тЖТ   255тЖТ
   256тЖТ   256тЖТ    # MAP estimate of Beta distribution
   257тЖТ   257тЖТ    if alpha + beta_p > 2:
   258тЖТ   258тЖТ        map_estimate = (alpha - 1) / (alpha + beta_p - 2)
   259тЖТ   259тЖТ    else:
   260тЖТ   260тЖТ        map_estimate = alpha / (alpha + beta_p)
   261тЖТ   261тЖТ
   262тЖТ   262тЖТ    return max(0.0, min(1.0, map_estimate))

   265тЖТ   265тЖТdef calculate_score_confidence(history, prior_alpha=2.0, prior_beta=2.0):
   266тЖТ   266тЖТ    """
   267тЖТ   267тЖТ    Calculate confidence in the trust score based on sample size.
   268тЖТ   268тЖТ    Uses variance of the Beta posterior distribution.
   269тЖТ   269тЖТ    
   270тЖТ   270тЖТ    More runs = higher confidence = we trust the score more.
   271тЖТ   271тЖТ    Fewer runs = lower confidence = score is more uncertain.
   272тЖТ   272тЖТ    
   273тЖТ   273тЖТ    Returns 0.0 to 1.0
   274тЖТ   274тЖТ    """
   275тЖТ   275тЖТ    if not history:
   276тЖТ   276тЖТ        return 0.0
   277тЖТ   277тЖТ
   278тЖТ   278тЖТ    passed = sum(1 for h in history if h["status"] == "passed")
   279тЖТ   279тЖТ    failed = len(history) - passed
   280тЖТ   280тЖТ
   281тЖТ   281тЖТ    alpha = prior_alpha + passed
   282тЖТ   282тЖТ    beta_p = prior_beta + failed
   283тЖТ   283тЖТ
   284тЖТ   284тЖТ    # Variance of Beta distribution
   285тЖТ   285тЖТ    n = alpha + beta_p
   286тЖТ   286тЖТ    if n <= 0:
   287тЖТ   287тЖТ        return 0.0
   288тЖТ   288тЖТ
   289тЖТ   289тЖТ    variance = (alpha * beta_p) / (n * n * (n + 1))
   290тЖТ   290тЖТ
   291тЖТ   291тЖТ    # Standard deviation
   292тЖТ   292тЖТ    std = math.sqrt(variance)
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ    # Convert to confidence (lower std = higher confidence)
   295тЖТ   295тЖТ    # std of Beta(2,2) is ~0.195 тЖТ confidence ~0.3
   296тЖТ   296тЖТ    # std of Beta(20,2) is ~0.084 тЖТ confidence ~0.7
   297тЖТ   297тЖТ    # std of Beta(100,10) is ~0.038 тЖТ confidence ~0.9
   298тЖТ   298тЖТ    confidence = max(0.0, min(1.0, 1.0 - std * 3.5))
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ    return round(confidence, 3)

   307тЖТ   307тЖТdef wilson_score_interval(history, confidence=0.95):
   308тЖТ   308тЖТ    """
   309тЖТ   309тЖТ    Wilson score interval тАФ better than normal approximation for proportions.
   310тЖТ   310тЖТ    Returns (lower_bound, upper_bound, point_estimate).
   311тЖТ   311тЖТ    """
   312тЖТ   312тЖТ    if not history:
   313тЖТ   313тЖТ        return (0.25, 0.75, 0.5)
   314тЖТ   314тЖТ
   315тЖТ   315тЖТ    n = len(history)
   316тЖТ   316тЖТ    p_hat = sum(1 for h in history if h["status"] == "passed") / n
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ    z = 1.96  # 95% confidence
   319тЖТ   319тЖТ    denom = 1 + z * z / n
   320тЖТ   320тЖТ    center = (p_hat + z * z / (2 * n)) / denom
   321тЖТ   321тЖТ    margin = z * math.sqrt((p_hat * (1 - p_hat) + z * z / (4 * n)) / n) / denom
   322тЖТ   322тЖТ
   323тЖТ   323тЖТ    lower = max(0.0, center - margin)
   324тЖТ   324тЖТ    upper = min(1.0, center + margin)
   325тЖТ   325тЖТ
   326тЖТ   326тЖТ    return (round(lower, 4), round(upper, 4), round(p_hat, 4))

   333тЖТ   333тЖТdef calculate_duration_variance(history):
   334тЖТ   334тЖТ    """Coefficient of variation of test durations."""
   335тЖТ   335тЖТ    if len(history) < 3:
   336тЖТ   336тЖТ        return 0.0
   337тЖТ   337тЖТ    durations = [h["duration"] for h in history if h["duration"] and h["duration"] > 0]
   338тЖТ   338тЖТ    if len(durations) < 3:
   339тЖТ   339тЖТ        return 0.0
   340тЖТ   340тЖТ    mean = sum(durations) / len(durations)
   341тЖТ   341тЖТ    variance = sum((d - mean) ** 2 for d in durations) / len(durations)
   342тЖТ   342тЖТ    std = math.sqrt(variance)
   343тЖТ   343тЖТ    cv = std / mean if mean > 0 else 0
   344тЖТ   344тЖТ    return min(cv * 100, 100)

   347тЖТ   347тЖТdef calculate_duration_zscores(history):
   348тЖТ   348тЖТ    """
   349тЖТ   349тЖТ    Calculate z-scores for each duration to detect outliers.
   350тЖТ   350тЖТ    Returns list of (duration, z_score, is_anomaly) tuples.
   351тЖТ   351тЖТ    An anomaly = |z_score| > 2.0 (outside 95% of distribution).
   352тЖТ   352тЖТ    """
   353тЖТ   353тЖТ    durations = [h["duration"] for h in history if h["duration"] and h["duration"] > 0]
   354тЖТ   354тЖТ    if len(durations) < 3:
   355тЖТ   355тЖТ        return []
   356тЖТ   356тЖТ
   357тЖТ   357тЖТ    mean = sum(durations) / len(durations)
   358тЖТ   358тЖТ    variance = sum((d - mean) ** 2 for d in durations) / len(durations)
   359тЖТ   359тЖТ    std = math.sqrt(variance)
   360тЖТ   360тЖТ
   361тЖТ   361тЖТ    if std == 0:
   362тЖТ   362тЖТ        return [(d, 0.0, False) for d in durations]
   363тЖТ   363тЖТ
   364тЖТ   364тЖТ    results = []
   365тЖТ   365тЖТ    for d in durations:
   366тЖТ   366тЖТ        z = (d - mean) / std
   367тЖТ   367тЖТ        is_anomaly = abs(z) > 2.0
   368тЖТ   368тЖТ        results.append((d, round(z, 2), is_anomaly))
   369тЖТ   369тЖТ
   370тЖТ   370тЖТ    return results

   417тЖТ   417тЖТ    How consistent are error messages when the test fails.
   418тЖТ   418тЖТ    High consistency = same error every time = might be a real bug, not flaky.
   419тЖТ   419тЖТ    Low consistency = different errors = likely non-deterministic / flaky.
   420тЖТ   420тЖТ    
   421тЖТ   421тЖТ    Returns 0-100 where 100 = perfectly consistent errors.
   422тЖТ   422тЖТ    """
   423тЖТ   423тЖТ    if not history:
   424тЖТ   424тЖТ        return 0.0
   425тЖТ   425тЖТ
   426тЖТ   426тЖТ    errors = [h["error_message"] for h in history if h["status"] == "failed" and h["error_message"]]
   427тЖТ   427тЖТ    if len(errors) < 2:
   428тЖТ   428тЖТ        return 0.0
   429тЖТ   429тЖТ
   430тЖТ   430тЖТ    # Normalize error messages: strip whitespace, lowercase, remove line numbers
   431тЖТ   431тЖТ    def normalize(err):
   432тЖТ   432тЖТ        import re
   433тЖТ   433тЖТ        err = err.strip().lower()
   434тЖТ   434тЖТ        # Remove common variable parts like line numbers, file paths
   435тЖТ   435тЖТ        err = re.sub(r'line \d+', 'line N', err)
   436тЖТ   436тЖТ        err = re.sub(r'at .+:\d+', 'at FILE:LINE', err)
   437тЖТ   437тЖТ        err = re.sub(r'0x[0-9a-f]+', '0xADDR', err)
   438тЖТ   438тЖТ        return err
   439тЖТ   439тЖТ
   440тЖТ   440тЖТ    normalized = [normalize(e) for e in errors]
   441тЖТ   441тЖТ    error_hashes = [hashlib.md5(e.encode()).hexdigest() for e in normalized]
   442тЖТ   442тЖТ    counter = Counter(error_hashes)
   443тЖТ   443тЖТ    most_common_ratio = counter.most_common(1)[0][1] / len(error_hashes)
   444тЖТ   444тЖТ
   445тЖТ   445тЖТ    return most_common_ratio * 100
   446тЖТ   446тЖТ

   449тЖТ   449тЖТ    """Extract the top N unique error patterns with frequencies."""
   450тЖТ   450тЖТ    if not history:
   451тЖТ   451тЖТ        return []
   452тЖТ   452тЖТ
   453тЖТ   453тЖТ    errors = [h["error_message"] for h in history if h["status"] == "failed" and h["error_message"]]
   454тЖТ   454тЖТ    if not errors:
   455тЖТ   455тЖТ        return []
   456тЖТ   456тЖТ
   457тЖТ   457тЖТ    def normalize(err):
   458тЖТ   458тЖТ        import re

   464тЖТ   464тЖТ

   483тЖТ   483тЖТdef detect_sequential_pattern(history):
   484тЖТ   484тЖТ    """
   485тЖТ   485тЖТ    Detect if pass/fail follows a sequential pattern (alternating, periodic, etc.).
   486тЖТ   486тЖТ    Returns pattern type or None.
   487тЖТ   487тЖТ    
   488тЖТ   488тЖТ    Checks for:
   489тЖТ   489тЖТ    - Alternating: P-F-P-F or F-P-F-P
   490тЖТ   490тЖТ    - Periodic: P-P-F-P-P-F (regular interval failures)
   491тЖТ   491тЖТ    - Clustered: P-P-P-F-F-F (sudden shift in behavior)
   492тЖТ   492тЖТ    """
   493тЖТ   493тЖТ    if len(history) < 6:
   494тЖТ   494тЖТ        return None
   495тЖТ   495тЖТ
   496тЖТ   496тЖТ    statuses = [1 if h["status"] == "passed" else 0 for h in history]
   497тЖТ   497тЖТ    n = len(statuses)
   498тЖТ   498тЖТ
   499тЖТ   499тЖТ    # Alternating check
   500тЖТ   500тЖТ    transitions = sum(1 for i in range(n - 1) if statuses[i] != statuses[i + 1])
   501тЖТ   501тЖТ    transition_rate = transitions / (n - 1)
   502тЖТ   502тЖТ
   503тЖТ   503тЖТ    if transition_rate > 0.7:
   504тЖТ   504тЖТ        return "alternating"
   505тЖТ   505тЖТ
   506тЖТ   506тЖТ    # Clustered check (runs test - Wald-Wolfowitz)
   507тЖТ   507тЖТ    runs = 1
   508тЖТ   508тЖТ    for i in range(1, n):
   509тЖТ   509тЖТ        if statuses[i] != statuses[i - 1]:
   510тЖТ   510тЖТ            runs += 1
   511тЖТ   511тЖТ
   512тЖТ   512тЖТ    expected_runs = 1 + (2 * sum(statuses) * (n - sum(statuses))) / n
   513тЖТ   513тЖТ    if n > 1:
   514тЖТ   514тЖТ        runs_variance = (2 * sum(statuses) * (n - sum(statuses)) * (2 * sum(statuses) * (n - sum(statuses)) - n)) / (n * n * (n - 1))
   515тЖТ   515тЖТ        if runs_variance > 0:
   516тЖТ   516тЖТ            z = (runs - expected_runs) / math.sqrt(runs_variance)
   517тЖТ   517тЖТ            if z < -1.96:  # Fewer runs than expected = clustered
   518тЖТ   518тЖТ                return "clustered"
   519тЖТ   519тЖТ
   520тЖТ   520тЖТ    # Periodic check (autocorrelation at lag 2, 3, 4, 5)
   521тЖТ   521тЖТ    mean_s = sum(statuses) / n
   522тЖТ   522тЖТ    if mean_s == 0 or mean_s == 1:
   523тЖТ   523тЖТ        return None
   524тЖТ   524тЖТ
   525тЖТ   525тЖТ    variance = sum((s - mean_s) ** 2 for s in statuses) / n
   526тЖТ   526тЖТ    if variance == 0:
   527тЖТ   527тЖТ        return None
   528тЖТ   528тЖТ
   529тЖТ   529тЖТ    for lag in range(2, min(6, n // 2)):
   530тЖТ   530тЖТ        autocorr = sum((statuses[i] - mean_s) * (statuses[i + lag] - mean_s) for i in range(n - lag)) / (n * variance)
   531тЖТ   531тЖТ        if autocorr > 0.5:
   532тЖТ   532тЖТ            return f"periodic_{lag}"
   533тЖТ   533тЖТ
   534тЖТ   534тЖТ    return None
   535тЖТ   535тЖТ

   541тЖТ   541тЖТdef analyze_environment_correlation(history):
   542тЖТ   542тЖТ    """
   543тЖТ   543тЖТ    Check if test outcomes correlate with specific environments.
   544тЖТ   544тЖТ    Returns dict with environment-specific pass rates if significant difference found.
   545тЖТ   545тЖТ    """
   546тЖТ   546тЖТ    env_groups = defaultdict(lambda: {"passed": 0, "total": 0, "durations": []})
   547тЖТ   547тЖТ
   548тЖТ   548тЖТ    for h in history:
   549тЖТ   549тЖТ        env = h.get("environment") or "unknown"
   550тЖТ   550тЖТ        env_groups[env]["total"] += 1
   551тЖТ   551тЖТ        if h["status"] == "passed":
   552тЖТ   552тЖТ            env_groups[env]["passed"] += 1
   553тЖТ   553тЖТ        if h["duration"]:
   554тЖТ   554тЖТ            env_groups[env]["durations"].append(h["duration"])
   555тЖТ   555тЖТ
   556тЖТ   556тЖТ    if len(env_groups) < 2:
   557тЖТ   557тЖТ        return None
   558тЖТ   558тЖТ
   559тЖТ   559тЖТ    rates = {}
   560тЖТ   560тЖТ    for env, data in env_groups.items():
   561тЖТ   561тЖТ        if data["total"] >= 2:
   562тЖТ   562тЖТ            rates[env] = {
   563тЖТ   563тЖТ                "pass_rate": data["passed"] / data["total"],
   564тЖТ   564тЖТ                "runs": data["total"],
   565тЖТ   565тЖТ                "avg_duration": sum(data["durations"]) / len(data["durations"]) if data["durations"] else 0,
   566тЖТ   566тЖТ            }
   567тЖТ   567тЖТ
   568тЖТ   568тЖТ    if len(rates) < 2:
   569тЖТ   569тЖТ        return None
   570тЖТ   570тЖТ
   571тЖТ   571тЖТ    # Check if there's significant variance between environments
   572тЖТ   572тЖТ    pass_rates = [r["pass_rate"] for r in rates.values()]
   573тЖТ   573тЖТ    max_diff = max(pass_rates) - min(pass_rates)

   585тЖТ   585тЖТdef classify_flaky_category(history, current_result=None):
   586тЖТ   586тЖТ    """
   587тЖТ   587тЖТ    Classify WHY a test is flaky based on multi-signal analysis.
   588тЖТ   588тЖТ    
   589тЖТ   589тЖТ    Categories:
   590тЖТ   590тЖТ    - timing: High duration variance, outliers detected by z-score
   591тЖТ   591тЖТ    - order_dependency: Alternating/periodic pass-fail patterns
   592тЖТ   592тЖТ    - shared_state: Consistent errors, clustered pattern
   593тЖТ   593тЖТ    - non_deterministic_data: Many different error messages, random failures
   594тЖТ   594тЖТ    - environment_specific: Fails only in specific environments
   595тЖТ   595тЖТ    
   596тЖТ   596тЖТ    v2 improvements:
   597тЖТ   597тЖТ    - Works with just 2+ runs (was 3+)
   598тЖТ   598тЖТ    - Uses z-score for timing detection
   599тЖТ   599тЖТ    - Uses sequential pattern detection
   600тЖТ   600тЖТ    - Considers environment correlation
   601тЖТ   601тЖТ    - Returns confidence alongside category
   602тЖТ   602тЖТ    """
   603тЖТ   603тЖТ    if not history or len(history) < 2:
   604тЖТ   604тЖТ        return None
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ    statuses = [h["status"] for h in history]
   607тЖТ   607тЖТ    pass_rate = sum(1 for s in statuses if s == "passed") / len(statuses)
   608тЖТ   608тЖТ
   609тЖТ   609тЖТ    # Not flaky if almost always passes or always fails
   610тЖТ   610тЖТ    if pass_rate > 0.92 or pass_rate < 0.08:
   611тЖТ   611тЖТ        return None
   612тЖТ   612тЖТ
   613тЖТ   613тЖТ    durations = [h["duration"] for h in history if h["duration"] and h["duration"] > 0]
   614тЖТ   614тЖТ    errors = [h["error_message"] for h in history if h.get("error_message")]
   615тЖТ   615тЖТ
   616тЖТ   616тЖТ    scores = {}  # category -> confidence score
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ    # тФАтФАтФА Timing detection (z-score based) тФАтФАтФА
   619тЖТ   619тЖТ    if len(durations) >= 3:
   620тЖТ   620тЖТ        anomaly_rate = duration_anomaly_rate(history)
   621тЖТ   621тЖТ        cv = calculate_duration_variance(history) / 100  # normalize to 0-1
   622тЖТ   622тЖТ
   623тЖТ   623тЖТ        # High CV or high anomaly rate = timing issue
   624тЖТ   624тЖТ        timing_score = max(cv * 1.5, anomaly_rate * 2)
   625тЖТ   625тЖТ        if timing_score > 0.3:
   626тЖТ   626тЖТ            scores["timing"] = min(1.0, timing_score)
   627тЖТ   627тЖТ
   628тЖТ   628тЖТ    # тФАтФАтФА Sequential pattern detection тФАтФАтФА
   629тЖТ   629тЖТ    pattern = detect_sequential_pattern(history)
   630тЖТ   630тЖТ    if pattern == "alternating":
   631тЖТ   631тЖТ        scores["order_dependency"] = 0.85
   632тЖТ   632тЖТ    elif pattern == "clustered":
   633тЖТ   633тЖТ        scores["shared_state"] = 0.7
   634тЖТ   634тЖТ    elif pattern and pattern.startswith("periodic_"):
   635тЖТ   635тЖТ        scores["order_dependency"] = 0.6
   636тЖТ   636тЖТ
   637тЖТ   637тЖТ    # тФАтФАтФА Non-deterministic data detection тФАтФАтФА
   638тЖТ   638тЖТ    if len(errors) >= 2:
   639тЖТ   639тЖТ        unique_errors = len(set(errors))
   640тЖТ   640тЖТ        error_ratio = unique_errors / len(errors)
   641тЖТ   641тЖТ
   642тЖТ   642тЖТ        # Many different errors = non-deterministic
   643тЖТ   643тЖТ        if error_ratio > 0.5:
   644тЖТ   644тЖТ            scores["non_deterministic_data"] = min(1.0, error_ratio)
   645тЖТ   645тЖТ
   646тЖТ   646тЖТ    # тФАтФАтФА Shared state detection тФАтФАтФА
   647тЖТ   647тЖТ    if len(errors) >= 2:
   648тЖТ   648тЖТ        error_consistency = calculate_error_pattern(history) / 100
   649тЖТ   649тЖТ
   650тЖТ   650тЖТ        # Same error but intermittent = shared state corruption
   651тЖТ   651тЖТ        if error_consistency > 0.6 and 0.15 < pass_rate < 0.85:
   652тЖТ   652тЖТ            shared_score = error_consistency * (1 - abs(pass_rate - 0.5) * 0.5)
   653тЖТ   653тЖТ            if "shared_state" not in scores or shared_score > scores.get("shared_state", 0):
   654тЖТ   654тЖТ                scores["shared_state"] = shared_score
   655тЖТ   655тЖТ
   656тЖТ   656тЖТ    # тФАтФАтФА Environment-specific detection тФАтФАтФА
   657тЖТ   657тЖТ    env_corr = analyze_environment_correlation(history)
   658тЖТ   658тЖТ    if env_corr:
   659тЖТ   659тЖТ        scores["environment_specific"] = 0.8
   660тЖТ   660тЖТ
   661тЖТ   661тЖТ    # Pick highest confidence category
   662тЖТ   662тЖТ    if not scores:
   663тЖТ   663тЖТ        return None
   664тЖТ   664тЖТ
   665тЖТ   665тЖТ    best_category = max(scores, key=scores.get)
   666тЖТ   666тЖТ    return best_category

   673тЖТ   673тЖТdef calculate_trust_score(history):
   674тЖТ   674тЖТ    """
   675тЖТ   675тЖТ    Calculate 0-100 trust score for a test.
   676тЖТ   676тЖТ    
   677тЖТ   677тЖТ    Scoring formula (v2):
   678тЖТ   678тЖТ    - Bayesian Pass Rate (55% weight) тАФ more robust with small samples
   679тЖТ   679тЖТ    - Recent Trend (20% weight) тАФ is it getting better or worse?
   680тЖТ   680тЖТ    - Duration Stability (10% weight) тАФ consistent execution time
   681тЖТ   681тЖТ    - Error Pattern Score (10% weight) тАФ consistent errors = more trustworthy than random
   682тЖТ   682тЖТ    - Flaky Category Penalty (up to -20 points) тАФ based on category severity
   683тЖТ   683тЖТ    
   684тЖТ   684тЖТ    v2 improvements:
   685тЖТ   685тЖТ    - Bayesian pass rate instead of naive ratio
   686тЖТ   686тЖТ    - Z-score anomaly detection feeds into duration stability
   687тЖТ   687тЖТ    - Sequential pattern analysis for flaky detection
   688тЖТ   688тЖТ    - Confidence score returned alongside trust score
   689тЖТ   689тЖТ    - Better penalty scaling based on detection confidence
   690тЖТ   690тЖТ    """
   691тЖТ   691тЖТ    if not history:
   692тЖТ   692тЖТ        return 100.0
   693тЖТ   693тЖТ
   694тЖТ   694тЖТ    # 1. Bayesian Pass Rate (55%)
   695тЖТ   695тЖТ    bayesian_rate = calculate_bayesian_pass_rate(history)
   696тЖТ   696тЖТ    pass_score = bayesian_rate * 55
   697тЖТ   697тЖТ
   698тЖТ   698тЖТ    # 2. Recent Trend (20%)
   699тЖТ   699тЖТ    trend = calculate_recent_trend(history)
   700тЖТ   700тЖТ    # Scale trend: +50 trend = full 20pts, -50 trend = 0pts
   701тЖТ   701тЖТ    trend_score = max(0, (50 + trend)) / 100 * 20
   702тЖТ   702тЖТ
   703тЖТ   703тЖТ    # 3. Duration Stability (10%)
   704тЖТ   704тЖТ    duration_var = calculate_duration_variance(history)
   705тЖТ   705тЖТ    # Use z-score anomaly rate as additional signal
   706тЖТ   706тЖТ    anomaly_rate = duration_anomaly_rate(history)
   707тЖТ   707тЖТ    duration_penalty = min(duration_var, 100) * 0.07 + anomaly_rate * 30
   708тЖТ   708тЖТ    duration_score = max(0, 10 - duration_penalty)
   709тЖТ   709тЖТ
   710тЖТ   710тЖТ    # 4. Error Pattern Consistency (10%)
   711тЖТ   711тЖТ    error_pattern = calculate_error_pattern(history)
   712тЖТ   712тЖТ    # Higher consistency = slightly more trustworthy (we know what the error is)
   713тЖТ   713тЖТ    # But very high consistency with failures = real bug, not flaky
   714тЖТ   714тЖТ    error_score = (error_pattern / 100) * 10
   715тЖТ   715тЖТ
   716тЖТ   716тЖТ    # Combine
   717тЖТ   717тЖТ    score = pass_score + trend_score + duration_score + error_score
   718тЖТ   718тЖТ
   719тЖТ   719тЖТ    # 5. Flaky Category Penalty
   720тЖТ   720тЖТ    flaky_cat = classify_flaky_category(history, None)
   721тЖТ   721тЖТ    if flaky_cat:
   722тЖТ   722тЖТ        penalty_map = {
   723тЖТ   723тЖТ            "timing": 8,
   724тЖТ   724тЖТ            "order_dependency": 12,
   725тЖТ   725тЖТ            "shared_state": 15,
   726тЖТ   726тЖТ            "non_deterministic_data": 20,
   727тЖТ   727тЖТ            "environment_specific": 10,
   728тЖТ   728тЖТ        }
   729тЖТ   729тЖТ        score -= penalty_map.get(flaky_cat, 10)
   730тЖТ   730тЖТ
   731тЖТ   731тЖТ    return max(0, min(100, round(score, 1)))

   802тЖТ   802тЖТdef calculate_flaky_probability(history):
   803тЖТ   803тЖТ    """
   804тЖТ   804тЖТ    Return 0.0-1.0 probability that a test is flaky (not permanently broken).
   805тЖТ   805тЖТ    
   806тЖТ   806тЖТ    v2: Uses Bayesian pass rate and multiple signals.
   807тЖТ   807тЖТ    - Pass rate near 0 or 1 тЖТ not flaky (broken or stable)
   808тЖТ   808тЖТ    - Pass rate near 0.5 тЖТ high flaky probability
   809тЖТ   809тЖТ    - Multiple error patterns тЖТ higher flaky probability
   810тЖТ   810тЖТ    - Sequential patterns тЖТ higher flaky probability
   811тЖТ   811тЖТ    - High duration anomaly тЖТ higher flaky probability
   812тЖТ   812тЖТ    """
   813тЖТ   813тЖТ    if not history or len(history) < 2:
   814тЖТ   814тЖТ        return 0.0
   815тЖТ   815тЖТ
   816тЖТ   816тЖТ    bayesian_rate = calculate_bayesian_pass_rate(history)
   817тЖТ   817тЖТ
   818тЖТ   818тЖТ    # Extreme pass rates = not flaky
   819тЖТ   819тЖТ    if bayesian_rate < 0.05:
   820тЖТ   820тЖТ        return 0.0  # Broken, not flaky
   821тЖТ   821тЖТ    if bayesian_rate > 0.95:
   822тЖТ   822тЖТ        return 0.0  # Stable
   823тЖТ   823тЖТ
   824тЖТ   824тЖТ    # Base probability from distance from 0.5 (most uncertain at 0.5)
   825тЖТ   825тЖТ    base_prob = 1.0 - abs(bayesian_rate - 0.5) * 2
   826тЖТ   826тЖТ    base_prob = base_prob ** 0.8  # Flatten the curve slightly
   827тЖТ   827тЖТ
   828тЖТ   828тЖТ    # Error pattern modifier (many different errors = more likely flaky)
   829тЖТ   829тЖТ    error_pat = calculate_error_pattern(history) / 100
   830тЖТ   830тЖТ    error_modifier = 1.0 + (1.0 - error_pat) * 0.3  # Low consistency = higher probability
   831тЖТ   831тЖТ
   832тЖТ   832тЖТ    # Sequential pattern modifier
   833тЖТ   833тЖТ    seq_pattern = detect_sequential_pattern(history)
   834тЖТ   834тЖТ    seq_modifier = 1.0
   835тЖТ   835тЖТ    if seq_pattern == "alternating":
   836тЖТ   836тЖТ        seq_modifier = 1.3
   837тЖТ   837тЖТ    elif seq_pattern == "clustered":
   838тЖТ   838тЖТ        seq_modifier = 1.15
   839тЖТ   839тЖТ    elif seq_pattern and seq_pattern.startswith("periodic_"):
   840тЖТ   840тЖТ        seq_modifier = 1.2
   841тЖТ   841тЖТ
   842тЖТ   842тЖТ    # Duration anomaly modifier
   843тЖТ   843тЖТ    anomaly = duration_anomaly_rate(history)
   844тЖТ   844тЖТ    dur_modifier = 1.0 + anomaly * 0.5
   845тЖТ   845тЖТ
   846тЖТ   846тЖТ    # Confidence: more runs = more certain about flakiness
   847тЖТ   847тЖТ    confidence = calculate_score_confidence(history)
   848тЖТ   848тЖТ    conf_modifier = 0.5 + confidence * 0.5  # Scale 0.5-1.0
   849тЖТ   849тЖТ
   850тЖТ   850тЖТ    final_prob = base_prob * error_modifier * seq_modifier * dur_modifier * conf_modifier
   851тЖТ   851тЖТ
   852тЖТ   852тЖТ    return min(1.0, max(0.0, round(final_prob, 3)))
