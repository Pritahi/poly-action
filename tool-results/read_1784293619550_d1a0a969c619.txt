   814→// Draw chart on canvas
   815→function renderChart(history) {
   816→  const canvas = $('historyChart');
   817→  const container = canvas.parentElement;
   818→  const dpr = window.devicePixelRatio || 1;
   819→
   820→  canvas.width = container.clientWidth * dpr;
   821→  canvas.height = container.clientHeight * dpr;
   822→
   823→  const ctx = canvas.getContext('2d');
   824→  ctx.scale(dpr, dpr);
   825→
   826→  const W = container.clientWidth;
   827→  const H = container.clientHeight;
   828→  const pad = { top: 20, right: 20, bottom: 40, left: 44 };
   829→  const plotW = W - pad.left - pad.right;
   830→  const plotH = H - pad.top - pad.bottom;
   831→
   832→  // Clear
   833→  ctx.clearRect(0, 0, W, H);
   834→
   835→  // Background zones
   836→  const zoneRanges = [
   837→    { from: 80, to: 100, color: 'rgba(34,197,94,0.06)' },
   838→    { from: 50, to: 80, color: 'rgba(234,179,8,0.06)' },
   839→    { from: 0, to: 50, color: 'rgba(239,68,68,0.06)' }
   840→  ];
   841→
   842→  zoneRanges.forEach(z => {
   843→    const y1 = pad.top + plotH - (z.to / 100) * plotH;
   844→    const y2 = pad.top + plotH - (z.from / 100) * plotH;
   845→    ctx.fillStyle = z.color;
   846→    ctx.fillRect(pad.left, y1, plotW, y2 - y1);
   847→  });
   848→
   849→  // Grid lines
   850→  ctx.strokeStyle = 'rgba(39,39,42,0.6)';
   851→  ctx.lineWidth = 1;
   852→  for (let v = 0; v <= 100; v += 20) {
   853→    const y = pad.top + plotH - (v / 100) * plotH;
   854→    ctx.beginPath();
   855→    ctx.setLineDash([4, 4]);
   856→    ctx.moveTo(pad.left, y);
   857→    ctx.lineTo(W - pad.right, y);
   858→    ctx.stroke();
   859→    ctx.setLineDash([]);
   860→  }
   861→
   862→  // Y-axis labels
   863→  ctx.fillStyle = '#71717a';
   864→  ctx.font = '11px -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif';
   865→  ctx.textAlign = 'right';
   866→  ctx.textBaseline = 'middle';
   867→  for (let v = 0; v <= 100; v += 20) {
   868→    const y = pad.top + plotH - (v / 100) * plotH;
   869→    ctx.fillText(v, pad.left - 8, y);
   870→  }
   871→
   872→  if (!history.length) {
   873→    ctx.fillStyle = '#71717a';
   874→    ctx.textAlign = 'center';
   875→    ctx.font = '13px -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif';
   876→    ctx.fillText('No history data', W / 2, H / 2);
   877→    return;
   878→  }
   879→
   880→  // Take last 20
   881→  const pts = history.slice(-20);
   882→  const n = pts.length;
   883→
   884→  // X-axis labels
   885→  ctx.fillStyle = '#71717a';
   886→  ctx.textAlign = 'center';
   887→  ctx.textBaseline = 'top';
   888→  ctx.font = '10px -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif';
   889→
   890→  const xStep = n > 1 ? plotW / (n - 1) : 0;
   891→
   892→  pts.forEach((p, i) => {
   893→    const x = n > 1 ? pad.left + i * xStep : pad.left + plotW / 2;
   894→    // Show every other label if many points
   895→    if (n <= 10 || i % 2 === 0 || i === n - 1) {
   896→      ctx.fillText(i + 1, x, H - pad.bottom + 10);
   897→    }
   898→  });
   899→
   900→  // X-axis title
   901→  ctx.fillStyle = '#71717a';
   902→  ctx.font = '11px -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif';
   903→  ctx.fillText('Run #', W / 2, H - 6);
   904→
   905→  // Y-axis title
   906→  ctx.save();
   907→  ctx.translate(12, H / 2);
   908→  ctx.rotate(-Math.PI / 2);
   909→  ctx.fillText('Trust Score', 0, 0);
   910→  ctx.restore();
   911→
   912→  // Compute points
   913→  const coords = pts.map((p, i) => {
   914→    const x = n > 1 ? pad.left + i * xStep : pad.left + plotW / 2;
   915→    const score = Math.max(0, Math.min(100, p.trust_score || 0));
   916→    const y = pad.top + plotH - (score / 100) * plotH;
   917→    return { x, y, score };
   918→  });
   919→
   920→  // Area fill
   921→  if (coords.length > 1) {
   922→    ctx.beginPath();
   923→    ctx.moveTo(coords[0].x, pad.top + plotH);
   924→    coords.forEach(c => ctx.lineTo(c.x, c.y));
   925→    ctx.lineTo(coords[coords.length - 1].x, pad.top + plotH);
   926→    ctx.closePath();
   927→    const grad = ctx.createLinearGradient(0, pad.top, 0, pad.top + plotH);
   928→    grad.addColorStop(0, 'rgba(139,92,246,0.15)');
   929→    grad.addColorStop(1, 'rgba(139,92,246,0.0)');
   930→    ctx.fillStyle = grad;
   931→    ctx.fill();
   932→  }
   933→
   934→  // Line
   935→  ctx.beginPath();
   936→  ctx.strokeStyle = '#8b5cf6';
   937→  ctx.lineWidth = 2;
   938→  ctx.lineJoin = 'round';
   939→  ctx.lineCap = 'round';
   940→  coords.forEach((c, i) => {
   941→    if (i === 0) ctx.moveTo(c.x, c.y);
   942→    else ctx.lineTo(c.x, c.y);
   943→  });
   944→  ctx.stroke();
   945→
   946→  // Dots
   947→  coords.forEach(c => {
   948→    ctx.beginPath();
   949→    ctx.arc(c.x, c.y, 4, 0, Math.PI * 2);
   950→    ctx.fillStyle = trustColorHex(c.score);
   951→    ctx.fill();
   952→    ctx.strokeStyle = '#09090b';
   953→    ctx.lineWidth = 2;
   954→    ctx.stroke();
   955→  });
   956→}

   963→// Render run history table
   964→function renderHistory(history) {
   965→  window._chartHistory = history;
   966→  const reversed = [...history].reverse();
   967→  const body = $('historyBody');
   968→
   969→  body.innerHTML = reversed.map((h, i) => {
   970→    const num = reversed.length - i;
   971→    const status = h.status || 'unknown';
   972→    const duration = h.duration != null ? (h.duration < 1 ? (h.duration * 1000).toFixed(0) + 'ms' : h.duration.toFixed(2) + 's') : '—';
   973→    const error = h.error_message || '';
   974→    const score = Math.round(h.trust_score || 0);
   975→    const time = h.run_timestamp ? new Date(h.run_timestamp).toLocaleString('en-US', {
   976→      month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit'
   977→    }) : '—';
   978→
   979→    return `<tr>
   980→      <td style="color:var(--text-muted);font-size:12px">${num}</td>
   981→      <td><span class="status-dot ${status}"></span>${status.charAt(0).toUpperCase() + status.slice(1)}</td>
   982→      <td class="mono-cell">${duration}</td>
   983→      <td class="error-msg" title="Hover to expand">
   984→        ${error ? escapeHtml(error) : '<span style="color:var(--text-muted)">—</span>'}
   985→        ${error ? `<div class="error-tooltip">${escapeHtml(error)}</div>` : ''}
   986→      </td>
   987→      <td style="color:${trustColor(score)};font-weight:600">${score}</td>
   988→      <td style="color:var(--text-muted);font-size:12px;white-space:nowrap">${time}</td>
   989→    </tr>`;
   990→  }).join('');
   991→}

   993→// Render error patterns
   994→function renderErrorPatterns(history) {
   995→  const errors = history.filter(h => h.status === 'failed' && h.error_message).map(h => h.error_message);
   996→  if (errors.length < 2) return;
   997→
   998→  // Count unique errors (normalize for grouping)
   999→  const counts = {};
  1000→  errors.forEach(e => {
  1001→    // Normalize: take first 120 chars as key
  1002→    const key = e.trim().slice(0, 120);
  1003→    counts[key] = (counts[key] || 0) + 1;
  1004→  });
  1005→
  1006→  const sorted = Object.entries(counts).sort((a, b) => b[1] - a[1]).slice(0, 3);
  1007→
  1008→  if (!sorted.length) return;
  1009→
  1010→  $('errorPatternsSection').style.display = 'block';
  1011→  $('errorPatternsGrid').innerHTML = sorted.map(([msg, count]) => `
  1012→    <div class="error-pattern-card">
  1013→      <div class="error-pattern-text">${escapeHtml(msg)}</div>
  1014→      <div class="error-pattern-count">&times;${count}</div>
  1015→    </div>
  1016→  `).join('');
  1017→}

  1022→  div.textContent = str;
  1023→  return div.innerHTML;
  1024→}
