   835→// Trust score color
   836→function trustColor(score) {
   837→  if (score >= 90) return 'var(--green)';
   838→  if (score >= 70) return 'var(--yellow)';
   839→  if (score >= 50) return 'var(--orange)';
   840→  return 'var(--red)';

   843→// Format time
   844→function formatTime(ts) {
   845→  if (!ts) return '—';
   846→  const d = new Date(ts);
   847→  const now = new Date();
   848→  const diff = now - d;
   849→  if (diff < 60000) return 'just now';
   850→  if (diff < 3600000) return Math.floor(diff / 60000) + 'm ago';
   851→  if (diff < 86400000) return Math.floor(diff / 3600000) + 'h ago';
   852→  return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' });
   853→}

   984→      `<div class="run-dot ${r.status}"></div>`

   994→        </div>
   995→      </td>
   996→      <td>${passRate}</td>
   997→      <td>${cat ? `<span class="cat-tag ${catClass}">${catLabel}</span>` : '—'}</td>
   998→      <td><div class="run-dots">${dots || '<span style="color:var(--text-muted)">—</span>'}</div></td>
   999→      <td><a href="test-detail.html?repo_name=${encodeURIComponent(currentRepo)}&test_name=${testName}" class="view-btn" onclick="event.stopPropagation()">View</a></td>

  1005→function renderRuns(runs) {
  1006→  const body = $('runsBody');

  1009→  if (!runs.length) {

  1017→    const isAllPass = (r.failed || 0) === 0;
  1018→    const statusClass = isAllPass ? 'run-status-all-pass' : 'run-status-has-fail';
  1019→    const statusText = isAllPass ? 'All passed' : `${r.failed} failed`;
  1020→    const avgTrust = Math.round(r.avg_trust_score || 0);
  1021→    const runId = r.run_id || '';
  1022→    const truncId = runId.length > 24 ? runId.slice(0, 24) + '...' : runId;
  1023→
  1024→    return `<tr>
  1025→      <td class="run-id-cell" title="${runId}">${truncId}</td>
  1026→      <td><span class="branch-tag">${r.branch || '—'}</span></td>
  1027→      <td>${r.total_tests || 0}</td>
  1028→      <td>${r.passed || 0}</td>
  1029→      <td class="${statusClass}">${statusText}</td>
  1030→      <td style="color:${trustColor(avgTrust)};font-weight:600">${avgTrust}%</td>
  1031→      <td class="time-cell">${formatTime(r.timestamp)}</td>
  1032→    </tr>`;
  1033→  }).join('');
  1034→}
